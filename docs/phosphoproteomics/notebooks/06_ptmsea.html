  • Show All Code

Overview

Site-set enrichment over PTMsigDB v2.0.0 (mouse) using fgsea. PTMsigDB is the curated database that backs the original PTM-SEA method (Krug et al. 2019, Mol Cell Proteomics) — it bundles published phosphosite signatures for kinase activity (KINASE-), perturbations (PERT-) and disease states (DISEASE-).

For each contrast we rank phospho-sites by the MSstatsPTM Tvalue from the adjusted PTM table (notebook 02 → PTM_adjusted_GroupComparison.csv), so the rank reflects phospho-specific regulation that can’t be explained by changes in total protein abundance. A positive NES means the signature’s members are enriched at the top of the contrast’s ranking (numerator-side activation); a negative NES means the bottom (denominator-side).

Signed-set handling. Each PTMsigDB site member carries a direction tag (;u = expected up, ;d = expected down when the signature is active). The fgsea API doesn’t natively support signed sets, so we expand each site into two ranked entries — …;u carrying +Tvalue and …;d carrying -Tvalue. A signature with mixed ;u/;d members then receives a single, direction-aware NES from standard fgsea without any per-pathway looping. This reproduces the PTM-SEA scoring intent (the dataset agrees with the signature when up-sites are up and down-sites are down) inside an fgsea run.

Identifier format. PTMsigDB v2 ships GMTs in three identifier flavours (UniProt accession, ±7 flanking sequence, PSP site-group ID). Despite gene-symbol formats appearing in the description column, the actual signature members in the mouse …uniprot… GMT are keyed by UniProt accession (Q9WV60;S202-p;u). We match the contrast table directly in that format — no gene-symbol mapping is needed for the rank vector. Gene symbols are still resolved via org.Mm.eg.db for human-readable display in the leading-edge / CSV outputs.

Per-contrast signed rank vectors

For each (site × contrast) pair we keep the row with the largest |Tvalue| (some sites are reported by more than one peptide). Each retained site is then emitted twice into the rank vector — once tagged ;u (positive Tvalue) and once ;d (sign-flipped) — to enable PTM-SEA-style signed-set scoring under standard fgsea.

Code 
prep_ptmsea_ranks <- function(dt, contrast_label) {
  d <- dt[Label == contrast_label]
  d <- d[d[, .I[which.max(abs(Tvalue))], by = SiteId]$V1]
  vals <- c(setNames( d$Tvalue, paste0(d$SiteId, ";u")),
            setNames(-d$Tvalue, paste0(d$SiteId, ";d")))
  vals <- vals[is.finite(vals)]
  sort(vals, decreasing = TRUE)
}

prettify_leading_edge <- function(le_vec) {
  if (length(le_vec) == 0) return("")
  parsed <- str_match(le_vec, "^([^;]+);([STY][0-9]+)-p;([ud])$")
  acc  <- parsed[, 2]
  res  <- parsed[, 3]
  dir  <- parsed[, 4]
  gene <- acc2gene[acc, on = "Accession", x.Gene]
  gene <- ifelse(is.na(gene), acc, gene)
  paste(paste0(gene, "_", res, "(", dir, ")"), collapse = ",")
}

fgsea + barplots (mirrors notebook 05)

Code 
run_ptmsea <- function(ranks, comp_name, cat_name, pathways) {
  set.seed(42)
  fgsea_res <- fgsea(pathways    = pathways,
                     stats       = ranks,
                     minSize     = 5,
                     maxSize     = 500,
                     nPermSimple = 10000,
                     BPPARAM     = BiocParallel::MulticoreParam(workers = 4, progressbar = FALSE))
  if (nrow(fgsea_res) == 0) return(NULL)

  out <- fgsea_res %>%
    as_tibble() %>%
    mutate(leadingEdge = sapply(leadingEdge, prettify_leading_edge)) %>%
    arrange(padj)

  out_dir <- file.path(ptmsea_dir, comp_name)
  dir.create(out_dir, recursive = TRUE, showWarnings = FALSE)
  write.csv(out, file.path(out_dir, glue("PTMsigDB_{cat_name}.csv")),
            row.names = FALSE)

  top <- out %>%
    arrange(desc(NES)) %>% slice_head(n = 10) %>%
    bind_rows(out %>% arrange(NES) %>% slice_head(n = 10)) %>%
    distinct(pathway, .keep_all = TRUE) %>%
    mutate(sig_label = case_when(padj < 0.001 ~ "***",
                                  padj < 0.01  ~ "**",
                                  padj < 0.05  ~ "*",
                                  TRUE         ~ ""),
           pathway = factor(pathway,
                            levels = unique(pathway[order(NES)])))

  if (nrow(top) == 0) return(out)

  axis_zoom <- max(c(2.5, ceiling(max(abs(top$NES)) * 1.05)))
  comp_parts <- str_split(comp_name, "_vs_")[[1]]
  if (length(comp_parts) == 2) {
    label_A <- glue("{comp_parts[1]} Enriched")
    label_B <- glue("{comp_parts[2]} Enriched")
  } else {
    label_A <- "Numerator Enriched"
    label_B <- "Denominator Enriched"
  }

  p <- ggplot(top, aes(x = NES, y = pathway,
                       fill = ifelse(NES > 0, label_A, label_B))) +
    geom_col(width = 0.7) +
    geom_text(aes(label = sig_label,
                  hjust = ifelse(NES > 0, 1.2, -0.2)),
              color = "white", size = 5, vjust = 0.75, fontface = "bold") +
    scale_fill_manual(values = setNames(c("firebrick3", "navy"),
                                         c(label_A, label_B))) +
    labs(x = "Normalized Enrichment Score (NES)", y = NULL,
         title = glue("PTM-SEA — {gsub('_', ' ', comp_name)} ({cat_name})")) +
    theme_bw(base_size = 14) +
    theme(legend.position = "none",
          plot.title = element_text(hjust = 0.5, face = "bold", size = 14,
                                     margin = margin(b = 20)),
          panel.grid.major.y = element_blank()) +
    scale_y_discrete(labels = function(x) str_wrap(gsub("_", " ", x), width = 50)) +
    coord_cartesian(xlim = c(-axis_zoom, axis_zoom), clip = "off")

  ggsave(file.path(out_dir, glue("PTMsigDB_Barplot_{cat_name}.pdf")),
         p, width = 9, height = 7, device = cairo_pdf)
  print(p)
  out
}

walk2(names(gene_sets_master), gene_sets_master, function(cat_name, pathways) {
  walk2(contrasts_master, names(contrasts_master), function(ranks, comp_name) {
    run_ptmsea(ranks, comp_name, cat_name, pathways)
    gc()
  })
})

    scale_fill_manual(values = setNames(c("firebrick3", "navy"),
                                         c(label_A, label_B))) +
    labs(x = "Normalized Enrichment Score (NES)", y = NULL,
         title = glue("PTM-SEA — {gsub('_', ' ', comp_name)} ({cat_name})")) +
    theme_bw(base_size = 14) +
    theme(legend.position = "none",
          plot.title = element_text(hjust = 0.5, face = "bold", size = 14,
                                     margin = margin(b = 20)),
          panel.grid.major.y = element_blank()) +
    scale_y_discrete(labels = function(x) str_wrap(gsub("_", " ", x), width = 50)) +
    coord_cartesian(xlim = c(-axis_zoom, axis_zoom), clip = "off")

  ggsave(file.path(out_dir, glue("PTMsigDB_Barplot_{cat_name}.pdf")),
         p, width = 9, height = 7, device = cairo_pdf)
  print(p)
  out
}
